% Comando para producto tensorial
% \tensopt{A}{B} → A ⊗ B
% \tensopt{A}{B}[C] → (A ⊗ B) ⊗ C
% \tensopt{A}{B}[C][nopar] → A ⊗ B ⊗ C * El cuarto argumentos puede ser cualquier cosa
\NewDocumentCommand{\tensprod}{m m o o}{%
  \IfNoValueTF{#3}
    {#1 \mathbin{\mathop{\otimes}} #2}%
    {%
      \IfNoValueTF{#4}
        {\left(#1 \mathbin{\mathop{\otimes}} #2\right) \mathbin{\mathop{\otimes}} #3}%
        {#1 \mathbin{\mathop{\otimes}} #2 \mathbin{\mathop{\otimes}} #3}%
    }%
}
