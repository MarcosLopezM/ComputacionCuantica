% Comando para producto tensorial
% \tensopt{A}{B} → A ⊗ B
% \tensopt{A}{B}[C] → (A ⊗ B) ⊗ C
% \tensopt{A}{B}[C][nopar] → A ⊗ B ⊗ C * El cuarto argumentos puede ser cualquier cosa
\NewDocumentCommand{\tensprod}{m m o o}{%
  \IfNoValueTF{#3}
    {#1 \mathop{\otimes} #2}%
    {%
      \IfNoValueTF{#4}
        {\left(#1 \mathop{\otimes} #2\right) \mathop{\otimes} #3}% Con paréntesis
        {#1 \mathop{\otimes} #2 \mathop{\otimes} #3}% Sin paréntesis
    }%
}
